#define __SFR_OFFSET 0

#include "avr/io.h"

.global start
.global work

start:
LDI R16,0XFF
STS DDRD+0X20,R16  
RET

work:
L:LDI R17,0X81
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0XF3
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X49
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X61
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X33
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X25
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X05
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0XF1
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X01
STS PORTD+0X20,R17
CALL DELAY

LDI R17,0X21
STS PORTD+0X20,R17
CALL DELAY

RJMP L
RET

DELAY:
LDI R20,0XAA
L3:LDI R18,0XFF
L2:LDI R19,0XFF
L1:DEC R19
BRNE L1
DEC R18
BRNE L2
DEC R20
BRNE L3
RET
